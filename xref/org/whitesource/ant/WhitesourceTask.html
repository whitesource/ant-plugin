116     private void validateAndPrepare() {
117         // api key
118         if (StringUtils.isBlank(apiKey)) {
119             error("Missing API Key");
120         }
121 
122         // product
123         if (StringUtils.isBlank(product)) {
124             product = getProject().getName();
125         }
126 
127         // modules
128         if (modules.isEmpty()) {
129             Module module = new Module(); // Treat whole project as single module
130             module.setName("Default Module");
131             addDefaultPaths(module);
132             modules.add(module);
133         } else {
134             int emptyModules = 0;
135             for (Module module : modules) {
136                 if (StringUtils.isBlank(module.getName()) && StringUtils.isBlank(module.getToken())) {
137                     error("Expecting module name or token");
138                 }
139                 if (module.getPaths().isEmpty()) {
140                     emptyModules++;
141                 }
142             }
143             if (emptyModules > 0) {
144                 if (modules.size() == 1) {
145                     addDefaultPaths(modules.iterator().next());
146                 } else {
147                     error("Path not set");
148                 }
149             }
150         }
151 
152         // check policies
153         if (!checkPolicies.isEmpty()) {
154             shouldCheckPolicies = true;
155             policyCheck = checkPolicies.iterator().next();
156             File reportDir = policyCheck.getReportdir();
157             if (reportDir == null) {
158                 reportDir = new File(this.getProject().getBaseDir(), "reports");
159                 policyCheck.setReportdir(reportDir);
160             }
161             if (!reportDir.exists() && !reportDir.mkdirs()) {
162                 error("Policies report directory doesn't exists and can not be created");
163             }
164         }
165     }

167     private void addDefaultPaths(Module module) {
168         Project project = getProject();
169 
170         FileSet fs = new FileSet();
171         fs.setDir(project.getBaseDir());
172         List<String> includes = new ArrayList<String>();
173         for (String extension : Constants.DEFAULT_SCAN_EXTENSIONS) {
174             includes.add("**/*." + extension);
175         }
176         fs.setIncludes(StringUtils.join(includes, ","));
177 
178         Path path = new Path(project);
179         path.addFileset(fs);
180         module.addPath(path);
181     }

183     private void scanModules() {
184         log("Collecting OSS usage information");
185 
186         for (Module module : modules) {
187             // create project info
188             AgentProjectInfo projectInfo = new AgentProjectInfo();
189             if (StringUtils.isBlank(module.getName())) {
190                 projectInfo.setProjectToken(module.getToken());
191                 log("Processing module with token " + module.getToken());
192             } else {
193                 projectInfo.setCoordinates(new Coordinates(null, module.getName(), null));
194                 log("Processing " + module.getName());
195             }
196 
197             // get all files located in module paths
198             Set<File> filesToUpdate = new HashSet<File>(); // We're using a set in order to avoid chance of duplicate files.
199             for (Path path : module.getPaths()) {
200                 for (String includedFile : path.list()) {
201                     File file = new File(includedFile);
202                     if (!file.isDirectory()) {
203                         filesToUpdate.add(file);
204                     }
205                 }
206             }
207 
208             // calculate SHA-1 for all files
209             Collection<DependencyInfo> dependencies = projectInfo.getDependencies();
210             for (File file : filesToUpdate) {
211                 String fileName = file.getName();
212                 DependencyInfo dependency = new DependencyInfo();
213                 dependency.setArtifactId(fileName);
214                 dependency.setSystemPath(file.getAbsolutePath());
215                 try {
216                     dependency.setSha1(ChecksumUtils.calculateSHA1(file));
217                 } catch (IOException e) {
218                     log("Problem calculating SHA-1 for '" + fileName + "'", e, Project.MSG_DEBUG);
219                 }
220                 dependencies.add(dependency);
221             }
222 
223             projectInfos.add(projectInfo);
224             log("Found " + dependencies.size() + " direct dependencies");
225             debugAgentProjectInfos(projectInfos);
226         }
227     }

234     private void checkPolicies() {
235         if (shouldCheckPolicies) {
236             log("Checking policies");
237             try {
238                 CheckPoliciesResult result = service.checkPolicies(apiKey, product, productVersion, projectInfos);
239                 handlePoliciesResult(result);
240             } catch (WssServiceException e) {
241                 error(e);
242             }
243         }
244     }
245 
246     private void handlePoliciesResult(CheckPoliciesResult result) {
247         // generate report
248         try {
249             log("Creating policies report", Project.MSG_INFO);
250             PolicyCheckReport report = new PolicyCheckReport(result);
251             report.generate(policyCheck.getReportdir(), false);
252         } catch (IOException e) {
253             error(e);
254         }
255 
256         // handle rejections if any
257         if (result.hasRejections()) {
258             String rejectionsErrorMessage = "Some dependencies does not conform with open source policies, see report for details.";
259             if (policyCheck.isFailonrejection()) {
260                 throw new BuildException(rejectionsErrorMessage);
261             } else {
262                 log(rejectionsErrorMessage, Project.MSG_WARN);
263             }
264         } else {
265             log("All dependencies conform with open source policies");
266         }
267     }

269     private void updateInventory() {
270         log("Updating White Source");
271         try {
272             UpdateInventoryResult result = service.update(apiKey, product, productVersion, projectInfos);
273             logUpdateResult(result);
274         } catch (WssServiceException e) {
275             error("A problem occurred while updating projects: " + e.getMessage());
276         }
277     }
278 
279     private void logUpdateResult(UpdateInventoryResult result) {
280         log("White Source update results:");
281         log("White Source organization: " + result.getOrganization());
282 
283         // newly created projects
284         Collection<String> createdProjects = result.getCreatedProjects();
285         if (createdProjects.isEmpty()) {
286             log("No new projects found");
287         } else {
288             log(createdProjects.size() + " Newly created projects:");
289             for (String projectName : createdProjects) {
290                 log(projectName);
291             }
292         }
293 
294         // updated projects
295         Collection<String> updatedProjects = result.getUpdatedProjects();
296         if (updatedProjects.isEmpty()) {
297             log("No projects were updated");
298         } else {
299             log(updatedProjects.size() + " existing projects were updated:");
300             for (String projectName : updatedProjects) {
301                 log(projectName);
302             }
303         }
304     }

322     private void debugAgentProjectInfos(Collection<AgentProjectInfo> projectInfos) {
323         log("----------------- dumping projectInfos -----------------", Project.MSG_DEBUG);
324         log("Total number of projects : " + projectInfos.size(), Project.MSG_DEBUG);
325         for (AgentProjectInfo projectInfo : projectInfos) {
326             log("Project coordinates: " + projectInfo.getCoordinates(), Project.MSG_DEBUG);
327             log("Project parent coordinates: " + projectInfo.getParentCoordinates(), Project.MSG_DEBUG);
328             log("Project project token: " + projectInfo.getProjectToken(), Project.MSG_DEBUG);
329 
330             Collection<DependencyInfo> dependencies = projectInfo.getDependencies();
331             log("total # of dependencies: " + dependencies.size(), Project.MSG_DEBUG);
332             for (DependencyInfo info : dependencies) {
333                 log(info + " SHA-1: " + info.getSha1(), Project.MSG_DEBUG);
334             }
335         }
336         log("----------------- dump finished -----------------", Project.MSG_DEBUG);
337     }
